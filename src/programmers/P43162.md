## DFS
```
class Solution {
    
    // 각 컴퓨터 방문 여부 저장 배열
    boolean[] visited;
    
    public int solution(int n, int[][] computers) {
        int answer = 0;
        
        visited = new boolean[n];
        
        // 모든 컴퓨터 하나씩 확인
        for (int i = 0; i < n; i++) {
            if (!visited[i]) {
                dfs(i, computers);
                answer++;
            }
        }
        
        return answer;
    }
    
    // DFS
    void dfs(int node, int[][] computers) {
        // 현재 컴퓨터 방문 처리
        visited[node] = true;
        
        // 현재 컴퓨터와 연결된 다른 컴퓨터 탐색
        for (int i = 0; i < computers.length; i++) {
            // 연결되어 있고 아직 방문하지 않았다면
            if (computers[node][i] == 1 && !visited[i]) {
                // 재귀 호출
                dfs(i, computers);
            }
        }
    }
}

## BFS
```
import java.util.*;

class Solution {
    
    // 각 컴퓨터 방문 여부 저장 배열
    boolean[] visited;
    
    public int solution(int n, int[][] computers) {
        int answer = 0;
        
        visited = new boolean[n];
        
        // 모든 컴퓨터 하나씩 확인
        for (int i = 0; i < n; i++) {
            if (!visited[i]) {
                bfs(i, computers);
                answer++;
            }
        }
        
        return answer;
    }
    
    // BFS
    void bfs(int node, int[][] computers) {
        Queue<Integer> q = new LinkedList<>();
        // 시작 컴퓨터를 큐에 추가
        q.offer(node);
        // 방문 처리
        visited[node] = true;
        
        while (!q.isEmpty()) {
            // 현재 탐색 중인 컴퓨터
            int current = q.poll();
            
            // 현재 컴퓨터와 연결된 다른 컴퓨터 탐색
            for (int i = 0; i < computers.length; i++) {
                // 연결되어 있고 아직 방문하지 않았다면
                if (computers[current][i] == 1 && !visited[i]) {
                    // 큐에 추가하여 다음 탐색 대상으로 설정
                    q.offer(i);
                    // 방문 처리
                    visited[i] = true;
                }
            }
        }
    }
}
