## DFS
```
class Solution {

    int[] numbers;
    int target;
    int answer;
    
    public int solution(int[] numbers, int target) {
        answer = 0;
        this.numbers = numbers;
        this.target = target;
        
        dfs(0, 0);
        
        return answer;
    }
    
    void dfs(int index, int sum) {
        // 모든 숫자를 다 사용했다면
        if (index == numbers.length) {
            // 누적 합이 타겟과 같다면 answer++
            if (sum == target) answer++;
            return;
        }
        
        // 현재 숫자에 '+'
        dfs(index + 1, sum + numbers[index]);
        // 현재 숫자에 '-'
        dfs(index + 1, sum - numbers[index]);
    }
}
```


## BFS
```
import java.util.*;

public class P43165 {
	public int solution(int[] numbers, int target) {
		int answer = 0;
		
		Queue<Integer> q = new LinkedList<>();
		
        // 시작
		q.add(0);
		
        // numbers 숫자 하나씩 사용
		for (int number : numbers) {
			int size = q.size();
			
			for (int i = 0; i < size; i++) {
                // 이전 단계에서 계산된 값 꺼내기
				int current = q.poll();

                // 현재 숫자에 '+'
				q.add(current + number);

                // 현재 숫자에 '-'
				q.add(current - number);
			}
		}
		
        // 큐에 모든 경우의 수가 들어있음
		while (!q.isEmpty()) {
			if (q.poll() == target) answer++;
		}
		
		return answer;
	}
}
```
